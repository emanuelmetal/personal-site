```typescript
'use client';

import { ThemeProvider as NextThemesProvider } from 'next-themes';

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  return (
    <NextThemesProvider
      attribute="class"
      defaultTheme="system"
      enableSystem={true}
      disableTransitionOnChange={false}
      storageKey="theme"
    >
      {children}
    </NextThemesProvider>
  );
}
```

- `attribute="class"` — adds/removes `dark` class on `<html>` for Tailwind
- `defaultTheme="system"` — per D-10, detect system preference on first visit
- `enableSystem={true}` — per D-04, three-state includes system
- `storageKey="theme"` — per D-08, localStorage key

**Step 2: Create src/components/ThemeToggle.tsx** (new file)

Per D-02 (sun/moon icons), D-03 (icon represents current theme), D-04 (three-state cycle):

```typescript
'use client';

import { useState, useEffect } from 'react';
import { useTheme } from 'next-themes';
import { Sun, Moon, Monitor } from 'lucide-react';

export function ThemeToggle() {
  const [mounted, setMounted] = useState(false);
  const { theme, setTheme } = useTheme();

  useEffect(() => {
    setMounted(true);
  }, []);

  if (!mounted) {
    return (
      <button
        className="rounded-md p-2 w-10 h-10 text-gray-600 dark:text-gray-400"
        aria-label="Loading theme toggle"
        disabled
      >
        <div className="h-6 w-6" />
      </button>
    );
  }

  const cycleTheme = () => {
    if (theme === 'light') setTheme('dark');
    else if (theme === 'dark') setTheme('system');
    else setTheme('light');
  };

  const getIcon = () => {
    if (theme === 'dark') return Moon;
    if (theme === 'light') return Sun;
    return Monitor;
  };

  const Icon = getIcon();

  return (
    <button
      onClick={cycleTheme}
      className="rounded-md p-2 text-gray-600 hover:bg-gray-100 hover:text-gray-900 dark:text-gray-400 dark:hover:bg-gray-800 dark:hover:text-white focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-gray-900 dark:focus-visible:outline-white transition-colors"
      aria-label={`Theme: ${theme}. Click to cycle.`}
      title="Click to cycle: Light, Dark, System"
    >
      <Icon className="h-6 w-6" />
    </button>
  );
}
```

- Mounting guard prevents hydration mismatch (Pitfall 1 from research)
- Placeholder has same dimensions (w-10 h-10) to prevent layout shift
- Cycle order: light -> dark -> system -> light (per D-04)
- Per D-03: Sun shows in light mode, Moon in dark mode, Monitor in system mode
- Focus-visible outline for keyboard accessibility (WCAG AA)
- NOTE: Do NOT use useTranslations('theme') for aria-label yet — translation keys are added in this same task but i18n integration for the toggle label can use plain strings. The theme key names ('light', 'dark', 'system') are universal enough. If desired, useTranslations can be added, but ensure the ThemeToggle still works without being nested inside NextIntlClientProvider (it will be — ThemeProvider wraps outside).

Actually, since ThemeProvider wraps OUTSIDE NextIntlClientProvider in layout.tsx, and ThemeToggle is rendered inside Header which IS inside NextIntlClientProvider, useTranslations WILL work. So use it:

Replace the aria-label and title with translated versions using `useTranslations('theme')`:
